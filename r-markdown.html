# For the cross sectional data scenario, I will be plotting CO2 emissions in
# kilotons against GDP per capita. Region will be used as a category to
# visualise which region emits the most CO2 in the year 2020

# Data preparation for plotting in 1(b)
df_clean <- read.csv("WDI_clean.csv")

selected_cols <- c("Code","Country","Income.Group","Region","GDP_per_capita_USD","CO2_emissions_kt")

df_clean %>% filter(Year==2020) %>% 
  select(all_of(selected_cols)) %>% 
  na.omit()->
  df_cross_section

# Creating a basic scatter plot of CO2 emissions in kilotons against GDP per capita 
p <- ggplot(df_cross_section, aes(x=GDP_per_capita_USD, y=CO2_emissions_kt)) +
  geom_point(mapping = aes(color = Region))

# adding additional customization to scale the axes to log scale as the numbers are huge
# labeling of the plot with appropriate title and captions and adding 

p_CS <- p + scale_x_log10(labels = scales::dollar) + scale_y_log10(labels = scales::trans_format("log10", scales::math_format(10^.x))) +
  geom_smooth(method = "lm", se = FALSE) +
  labs(x = "GDP Per Capita (USD)", y = "CO2 Emissions (Kilotons)",
       title = "Economic Growth and CO2 Emissions",
       subtitle = "Data points are countries in the year 2020",
       color = "Region",
       caption = "Source: WDI") + theme_light() +
  theme(title = element_text(size=14), 
        axis.text.x = element_text(size = 12,margin = margin(b = 10)), 
        axis.text.y = element_text(size = 12,margin = margin(l = 10)),
        axis.title = element_text(size = 14),
        plot.title = element_text(margin = margin(b = 10)),
        legend.position ="bottom")

# Check the final plot 
p_CS
## `geom_smooth()` using formula = 'y ~ x'

# Finally, I export and save my plot to a png file.

ggsave("Economic Growth and CO2 Emissions.png",plot = p_CS)
